fn ordInsercion(arr: &mut [i64]) {
    let mut i: usize = 0;
    let mut j: usize = 0;
    let mut aux: i64 = 0;

    i = 1;
    while (i < (arr.len())) {
        j = i;
        aux = arr[i];

        while j > 0 && aux < arr[j-1] {
            arr[j] = arr[j - 1];
            j = j - 1;
        }
        
        arr[j] = aux;
        i = i + 1;
    }
}



fn main () {
let mut arr3: [i64; 6] = [90, 3, 40, 10, 8, 5];
    ordInsercion(&mut arr3);
    println!("{:?}", arr3);
  
    
}
